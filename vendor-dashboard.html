API Keys Configuration

API keys are stored securely in the server's .env file. Contact your administrator to update credentials.

Location: /api/.env 

# POD Vendors
PRINTFUL_API_KEY=your_key_here
PRINTIFY_API_KEY=your_key_here
PRINTIFY_SHOP_ID=your_shop_id

# Dropship Vendors
ALIEXPRESS_APP_KEY=your_key
ALIEXPRESS_APP_SECRET=your_secret
CJ_API_KEY=your_key
CJ_EMAIL=your_email

# Shipping
EASYPOST_API_KEY=your_key
SHIPSTATION_API_KEY=your_key
SHIPPO_API_KEY=your_key

# Payments
STRIPE_SECRET_KEY=sk_live_...
STRIPE_PUBLISHABLE_KEY=pk_live_...
